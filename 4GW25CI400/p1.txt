#include <stdio.h>
#define MAX 5

int queue[MAX];
int front = -1, rear = -1;

// Add GatePass request
void enqueue() {
    int passID;
    if (rear == MAX - 1) {
        printf("GatePass Queue is Full\n");
        return;
    }
    if (front == -1)
        front = 0;

    printf("Enter GatePass ID: ");
    scanf("%d", &passID);

    rear++;
    queue[rear] = passID;
    printf("GatePass %d Added\n", passID);
}

// Approve GatePass
void dequeue() {
    if (front == -1 || front > rear) {
        printf("No GatePass Requests\n");
        return;
    }
    printf("GatePass %d Approved\n", queue[front]);
    front++;
}

// Display Queue
void display() {
    if (front == -1 || front > rear) {
        printf("Queue is Empty\n");
        return;
    }
    printf("Pending GatePass IDs: ");
    for (int i = front; i <= rear; i++) {
        printf("%d ", queue[i]);
    }
    printf("\n");
}

int main() {
    int choice;
    do {
        printf("\n--- GatePass System ---\n");
        printf("1. Add GatePass\n");
        printf("2. Approve GatePass\n");
        printf("3. Display Pending GatePass\n");
        printf("4. Exit\n");
        printf("Enter choice: ");
        scanf("%d", &choice);

        switch (choice) {
            case 1: enqueue(); break;
            case 2: dequeue(); break;
            case 3: display(); break;
            case 4: printf("Exiting System\n"); break;
            default: printf("Invalid Choice\n");
        }
    } while (choice != 4);

    return 0;
}
